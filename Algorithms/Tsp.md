def promising (i, W, vindex):
	flag = True
	if ((i == (n - 1)) and not W[vindex[n-1]][vindex[0]]):
		flag = False
	elif ((i > 0) and not W[vindex[i-1]][vindex[i]]):
		flag = False
	else:
		j = 1
		while (j < i and flag):
			if (vindex[i] == vindex[j]):
				flag = False
			j += 1
	return flag
