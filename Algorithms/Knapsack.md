```python
def knapsack1(W, w, p):
	n = len(w) - 1
	K = [0] * (n + 1)
	weight = 0
	for i in range(1, n + 1):
		weight += w[i]
		K[i] = w[i]
		if (weight > W):
			K[i] -= (weight - W)
			break;
	return K
```
